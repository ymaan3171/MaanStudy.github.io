// server.js (Backend Entry Point) const express = require('express'); const app = express(); app.use(express.json()); // JSON data receive karne ke liye // Ek dummy database array (Reality mein yahan MongoDB Mongoose models honge) let testsDatabase = [ { id: 1, title: "RRB ALP (CBT 1 + CBT 2 + CBAT) Mock Test...", totalQuestions: 991 } ]; // ---------------------------------------------------- // USER ROUTES (User app inko fetch karegi UI dikhane ke liye) // ---------------------------------------------------- app.get('/api/user/tests', (req, res) => { // User jab login karega, uske enrolled tests database se nikal kar bhejenge res.json(testsDatabase); }); // ---------------------------------------------------- // ADMIN ROUTES (Admin Panel dashboard yahan data bhejegha) // ---------------------------------------------------- // 1. ADD: Admin naya test add karta hai app.post('/api/admin/tests', (req, res) => { const newTest = { id: testsDatabase.length + 1, title: req.body.title, totalQuestions: req.body.totalQuestions }; testsDatabase.push(newTest); res.status(201).json({ message: "New test added successfully by Admin!", test: newTest }); }); // 2. UPDATE: Admin purane test ko edit karta hai app.put('/api/admin/tests/:id', (req, res) => { const testId = parseInt(req.params.id); const testIndex = testsDatabase.findIndex(t => t.id === testId); if(testIndex !== -1) { testsDatabase[testIndex] = { ...testsDatabase[testIndex], ...req.body }; res.json({ message: "Test updated successfully!", test: testsDatabase[testIndex] }); } else { res.status(404).json({ message: "Test not found!" }); } }); // 3. DELETE: Admin test ko delete karta hai app.delete('/api/admin/tests/:id', (req, res) => { const testId = parseInt(req.params.id); testsDatabase = testsDatabase.filter(t => t.id !== testId); res.json({ message: "Test deleted successfully!" }); }); // Start Server app.listen(3000, () => { console.log("Server is running on port 3000"); });
